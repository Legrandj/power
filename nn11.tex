\begin{tikzpicture}
\begin{scope}[every node/.style={circle,thick,draw}]
    \node (A) at (0,0) {$x$};
    \node (D) at (6,0) {$y$};
\end{scope}

    \node (C) at (4,0) {$\sigma$};

              every edge/.style={draw=red,very thick}]
    \path [->] (A) edge node {}(B);
    \path [->] (B) edge node {}(C);
    \path [->] (C) edge node {} (D);
\end{scope}
\end{tikzpicture}
